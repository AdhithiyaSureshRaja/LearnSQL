    - Syntax: DATEPART(unit, date)

Logical and Control Flow Functions

1. CASE WHEN
    - Syntax: CASE WHEN condition THEN result [WHEN ...] [ELSE result] END
    - Example: SELECT CASE WHEN salary > 50000 THEN 'High' ELSE 'Low' END AS salary_level FROM employees;
2. COALESCE()
    - Syntax: COALESCE(value1, value2, ...)
    - Example: SELECT COALESCE(phone_number, email) AS contact_info FROM employees;
3. NULLIF()
    - Syntax: NULLIF(value1, value2)
    - Example: SELECT NULLIF(salary, 0) AS salary FROM employees;

Window Functions

1. RANK()
    - Syntax: RANK() OVER (PARTITION BY column ORDER BY column)
    - Example: SELECT RANK() OVER (PARTITION BY department ORDER BY salary DESC) AS salary_rank FROM employees;
2. DENSE_RANK()
    - Syntax: DENSE_RANK() OVER (PARTITION BY column ORDER BY column)
    - Example: SELECT DENSE_RANK() OVER (PARTITION BY department ORDER BY salary DESC) AS salary_rank FROM employees;
3. ROW_NUMBER()
    - Syntax: ROW_NUMBER() OVER (PARTITION BY column ORDER BY column)
    - Example: SELECT ROW_NUMBER() OVER (PARTITION BY department ORDER BY salary DESC) AS row_num FROM employees;
4. LAG()
    - Syntax: LAG(column, offset) OVER (PARTITION BY column ORDER BY column)
    - Example: SELECT LAG(salary, 1) OVER (PARTITION BY department ORDER BY salary DESC) AS prev_salary FROM employees;
5. LEAD()
    - Syntax: LEAD(column, offset) OVER (PARTITION BY column ORDER BY column)
    - Example: SELECT LEAD(salary, 1) OVER (PARTITION BY department ORDER BY salary DESC) AS next_salary FROM employees;
